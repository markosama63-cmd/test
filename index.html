import tkinter as tk from tkinter import ttk, messagebox import matplotlib.pyplot as plt from matplotlib.backends.backend_tkagg import FigureCanvasTkAgg import numpy as np class Process: """Process class representing a single process""" def _init_(self, pid, arrival, burst, priority=0): self.pid = pid self.arrival = arrival self.burst = burst self.remaining = burst # Reset all processes for p in processes: p.reset() time = 0 completed = 0 n = len(processes) ready_queue = [] gantt = [] last_process = None start_time = 0 # Sort by arrival time for initial processing processes.sort(key=lambda x: x.arrival) while completed < n: # Add all arrived processes to ready queue for p in processes: if p.arrival <= time and p.remaining > 0 and p not in ready_queue: ready_queue.append(p) if ready_queue: # Sort by remaining time (shortest first) ready_queue.sort(key=lambda x: x.remaining) current = ready_queue[0] # Set response time if first time if current.response_time == -1: current.response_time = time - current.arrival # Record Gantt chart if last_process != current: if last_process is not None: gantt.append((last_process, start_time, time)) start_time = time last_process = current # Execute for 1 unit current.remaining -= 1 time += 1 # Check if process completed if current.remaining == 0: current.finish_time = time current.turnaround_time = current.finish_time - current.arrival current.waiting_time = current.turnaround_time - current.burst ready_queue.remove(current) completed += 1 else: # No process ready, CPU idle if last_process is not None: gantt.append((last_process, start_time, time)) last_process = None time += 1 # Add last segment if last_process is not None: gantt.append((last_process, start_time, time)) return gantt, processes class RoundRobinScheduler: """Round Robin Scheduler with configurable time quantum""" def _init_(self, quantum=4): self.quantum = quantum def simulate(self, processes): # Reset all processes for p in processes: p.reset() time = 0 completed = 0 n = len(processes) ready_queue = [] gantt = [] remaining_burst = {p: p.burst for p in processes} # Sort by arrival time processes.sort(key=lambda x: x.arrival) # Initialize with first process for p in processes: if p.arrival <= time and p not in ready_queue and remaining_burst[p] > 0: ready_queue.append(p) last_process = None start_time = 0 while completed < n: if ready_queue: current = ready_queue.pop(0) # Set response time if current.response_time == -1: current.response_time = time - current.arrival # Record Gantt if last_process != current: if last_process is not None: gantt.append((last_process, start_time, time)) start_time = time last_process = current # Execute for quantum or remaining time exec_time = min(self.quantum, remaining_burst[current]) time += exec_time remaining_burst[current] -= exec_time # Add newly arrived processes to queue for p in processes: if p.arrival <= time and p not in ready_queue and remaining_burst[p] > 0: ready_queue.append(p) #
